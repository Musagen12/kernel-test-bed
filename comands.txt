qemu-system-x86_64 \
  -kernel bzImage \
  -drive file=rootfs.img,format=raw \
  -append "console=ttyS0 nokaslr net.ifnames=0 root=/dev/sda rw" \
  -nographic \
  -m 2G \
  -netdev user,id=net0 \
  -device e1000,netdev=net0 \
  -serial mon:stdio
